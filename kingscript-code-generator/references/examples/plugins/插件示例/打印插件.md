## 2、事件方法
### 根据控件标识，对不同控件进行设置
```kingscript
beforeOutputWidget(evt: BeforeOutputWidgetEvent): void {

    //获取控件标识
    let widgetKey = evt.getWidgetKey();

    //隐藏控件
    if("Text" === widgetKey){
        evt.setHidden(true);
    }

    //绑定控件值
    if("Text1" === widgetKey){
        evt.bindField("bd_customer","billno");
    }

    //设置控件值
    if("Text2" === widgetKey){
        evt.setOutputValue(new TextField("xxx"));
    }

    //修改控件样式
    if("Text3" === widgetKey){
        evt.setStyle(StyleKey.FONT_COLOR,"#fb2323");
    }

    //设置图片控件值
    if("Image" === widgetKey){
        //地址必须是苍穹附件服务器地址，或者当前服务器能够访问的资源路径
        evt.setOutputValue(new ImageField("/ierp/xxx/xx.jpg"));
    }

    //表格控件动态合并第一行，第1列~第3列
    if("DataGrid" === widgetKey){
        let gridEvent = evt as BeforeOutputGridEvent;
        gridEvent.addMerge(MergeBlock.create(0,2,0,0));
    }
}
```

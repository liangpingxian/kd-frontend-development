## 1、插件示例
### 定义一个引入插件
```kingscript
import { BatchImportPlugin} from "@cosmic/bos-core/kd/bos/form/plugin/impt";
import { List, ArrayList } from "@cosmic/bos-script/java/util";

class MyImportPlugin extends BatchImportPlugin {
    getOverrideFieldsConfig(): List {
        return new ArrayList();
    }
    getDefaultKeyFields(): string {
        return '';
    }
    getDefaultLockUIs(): List {
        return new ArrayList();
    }
    getDefaultImportType(): string {
        return '';
    }
}

let plugin = new MyImportPlugin();

export { plugin }
```

## 2、事件方法
### 打开起始页时，自定义更新导入的匹配字段下拉框的下拉选项值
```kingscript
/**
 * 覆盖引入的匹配字段备选项
 * @returns 
 */
getOverrideFieldsConfig(): List {
    let items = new ArrayList();
    items.add(new ComboItem(new LocaleString("内码"), "id"));
    items.add(new ComboItem(new LocaleString("单据编号"), "billno"));
    return items;
}
```

### 打开起始页时，自定义更新导入的匹配字段下拉框的默认值
```kingscript
/**
 * 覆盖引入的匹配字段的缺省值
 * @returns 
 */
getDefaultKeyFields(): string {
    return 'id,billno';
}
```

### 打开起始页时，禁用导入下拉框
```kingscript
/**
 * 锁定的控件列表
 * @returns 
 */
getDefaultLockUIs(): List {
    let lockKeys = new ArrayList();
    // 锁定导入类型
    lockKeys.add("radiofield");
    lockKeys.add("radiofield1");
    return lockKeys;
}
```

### 打开起始页时，自定义默认的导入方式
```kingscript
/**
 * 缺省引入模式，包括新增、覆盖、覆盖并新增
 * @returns 
 */
getDefaultImportType(): string {
    return 'overridenew';
}
```

## 1、插件示例
### 定义一个转换插件
```kingscript
import { AbstractConvertPlugIn } from "@cosmic/bos-core/kd/bos/entity/botp/plugin";
import { AfterCreateTargetEventArgs, AfterFieldMappingEventArgs, AfterConvertEventArgs } from "@cosmic/bos-core/kd/bos/entity/botp/plugin/args";

class MyPlugin extends AbstractConvertPlugIn {

    /**
     * Event after creating the target document data packet: Pass the created target document according to the split rules to the plugin
     */
    afterCreateTarget(e: AfterCreateTargetEventArgs): void {
        super.afterCreateTarget(e);
        
    }

    /**
     * Event after target fields are assigned: The plugin can continue to fill in the target fields based on this
     */
    afterFieldMapping(e: AfterFieldMappingEventArgs): void {
        super.afterFieldMapping(e);
        
    }

    /**
     * Event after document conversion, executed last: The plugin can make final modifications to the generated target document data in this event
     */
    afterConvert(e: AfterConvertEventArgs): void {
        super.afterConvert(e);
        
    }

    
}

let plugin = new MyPlugin();

export { plugin };

```

initVariable(e: InitVariableEventArgs): void  {
    if (e.getConvertArgs() instanceof PushArgs) {
        let args = e.getConvertArgs() as PushArgs;
        let rowKey = 0;
        let rows = args.getSelectedRows();
        for (let i = 0; i < rows.size(); i++) {
            let row = rows.get(i)
            row.setPageIndex(1);
            row.setRowKey(rowKey);  // 调整单据下推顺序，rowKey小的放在前面
            rowKey++;
        }
    }
}

### 采购单据，转固定资产卡片时，每个物品生成一张开片，即按数量分单
```kingscript
afterConvert(e: AfterConvertEventArgs): void  {
    
    // 获取已生成的资产卡片
    let billDataEntitys = e.getTargetExtDataEntitySet().FindByEntityKey("fa_card_real");
    
    // 构造 ExtendedDataEntity 时需要的索引值
    let dataIndex = billDataEntitys.length;
    
    let copyDataEntitys = new ArrayList();
    for(let billDataEntity of billDataEntitys){
        
        // 原始的资产数量
        let qty = billDataEntity.getValue("assetamount");
        
        // 将资产数量改为1
        billDataEntity.setValue("assetamount", 1);
        
        // 来源分录拆分序号 从1开始
        let splitSeq = 1;
        billDataEntity.setValue("sourceentrysplitseq", splitSeq++);
        
        // 复制 （原始的资产数量 - 1）个卡片对象
        for(let i = 1; i < qty; i++){
            let copyObj = OrmUtils.clone(billDataEntity.getDataEntity(), false, true) as DynamicObject;
            copyObj.set("sourceentrysplitseq", splitSeq++);
            
            copyDataEntitys.add(new ExtendedDataEntity(copyObj, dataIndex++, 0));
        }
    }
    
    // 将复制出的单据，放入 targetExtDataEntitySet ，最终就会生成那么多的卡片
    e.getTargetExtDataEntitySet().AddExtendedDataEntities("fa_card_real", copyDataEntitys);
}
```

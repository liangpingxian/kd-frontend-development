## 1、插件示例
### 定义一个报表查询插件
```kingscript
import { DataSet } from "@cosmic/bos-core/kd/bos/algo";
import { AbstractReportListDataPlugin, ReportQueryParam } from "@cosmic/bos-core/kd/bos/entity/report";
import { QFilter } from "@cosmic/bos-core/kd/bos/orm/query";
import { QueryServiceHelper } from "@cosmic/bos-core/kd/bos/servicehelper";
import { List } from "@cosmic/bos-script/java/util";

class MyReportListDataPlugin extends AbstractReportListDataPlugin {

    query(queryParam: ReportQueryParam, selectedObj: any): DataSet {
        let filters: QFilter[] = [];
        return QueryServiceHelper.queryDataSet("bos_user", "bos_user", "id,name", filters, "id,asc", 100);;
    }

    getColumns(columns: List): List {
        return columns;
    }
}

let plugin = new MyReportListDataPlugin();

export { plugin };
```

## 2、事件方法
### 自行取数构建Dataset
```kingscript
/**
* 对数据列操作
* @param queryParam
* @param selectedObj
* @returns
*/
query(queryParam: ReportQueryParam, selectedObj: any): DataSet {
    // 根据过滤条件取数，构建DataSet

    // 获取表单插件设置的过滤条件
    let queryParamQFilters = queryParam.getFilter().getQFilters();
    let qFilters: QFilter[] = [];
    for (let i = 0; i < queryParamQFilters.size(); i++){
        qFilters.push(queryParamQFilters.get(i));
    }
    // 获取表单插件设置的排序条件
    let orderBys = queryParam.getSortInfo();
    // 取数
    return QueryServiceHelper.queryDataSet("kingscript", "kdtest_kingscript_purreq", "kdtest_applier, auditor, kdtest_totalamount, kdtest_applydate, billstatus", qFilters, orderBys, 100);
}
```

### 对报表指定列属性进行修改
```kingscript
/**
 * 对指定的列的属性进行修改
 * @param columns 
 * @returns 
 */
getColumns(columns: List): List {
    // 修改指定列的列名

    for(let i = 0; i < columns.size(); i++) {
        let rColumn = columns.get(i) as ReportColumn;
        let key = rColumn.getFieldKey();
        if("kdtest_totalamount" === key) {
            rColumn.setCaption(new LocaleString("申请单金额"));           
            columns.set(i, rColumn);
        }
    }
    return columns;
}
```

### 1. query
**定义**: `public abstract DataSet query(ReportQueryParam queryParam, Object selectedObj)`
**场景**: **核心方法**。实现报表的主取数逻辑，返回一个 `DataSet` 数据集。
```typescript
public query(queryParam: ReportQueryParam, selectedObj: any): DataSet {
    // 场景：从多个业务表汇总数据，构造报表结果
    const sql = "select fmaterial, fqty, famount from my_business_table where fdate >= ?";
    const date = queryParam.getFilterFieldValue("start_date");
    return DBServiceHelper.queryDataSet(this.getPluginName(), "id", sql, [date]);
}
```

### 2. getColumns
**定义**: `public List<AbstractReportColumn> getColumns(List<AbstractReportColumn> columns)`
**场景**: 动态定义报表的列。例如根据选中的维度（按仓库或按物料）动态生成列。
```typescript
public getColumns(columns: any): any {
    // 场景：根据用户勾选的展示维度，动态添加“批号”列
    if (this.getQueryParam().getCustomParam("show_batch")) {
        const col = new ReportColumn();
        col.setFieldKey("batchno");
        col.setCaption(new LocaleString("批号"));
        columns.add(col);
    }
    return columns;
}
```

### 3. export
**定义**: `public DataSet export(...)`
**场景**: 自定义引出（导出）时的取数逻辑。如果导出数据需要比界面显示更详细，可重写此方法。
```typescript
public export(queryParam: ReportQueryParam, selectedObj: any): DataSet {
    // 场景：引出时增加“备注”等长文本字段，界面为了性能默认不展示
    return this.query(queryParam, selectedObj); // 默认调用 query
}
```

## 工具方法（非事件）

### 1. getOperationResult - 获取操作结果

**签名**：`OperationResult getOperationResult()`

**说明**：获取当前操作的执行结果对象，包含成功/失败状态、错误信息、成功的主键ID集合等。

**业务场景**：审核操作完成后，根据操作结果判断是否需要触发后续流程。

```typescript
import { AbstractOperationServicePlugIn } from "@cosmic/bos-core/kd/bos/entity/plugin";

class AuditResultPlugin extends AbstractOperationServicePlugIn {

  afterExecuteOperationTransaction(e: AfterOperationArgs): void {
    super.afterExecuteOperationTransaction(e);

    // 获取操作结果
    let operationResult = this.getOperationResult();

    if (operationResult.isSuccess()) {
      // 获取审核成功的单据主键集合
      let successPkIds = operationResult.getSuccessPkIds();
      console.log("审核成功的单据数量：" + successPkIds.size());

      // 获取操作返回消息
      let message = operationResult.getMessage();
      console.log("操作结果消息：" + message);
    } else {
      // 获取所有错误信息
      let errorInfos = operationResult.getAllErrorOrValidateInfo();
      for (let i = 0; i < errorInfos.size(); i++) {
        let errorInfo = errorInfos.get(i);
        console.log("错误信息：" + errorInfo.getMessage());
      }
    }
  }
}

let plugin = new AuditResultPlugin();
export { plugin };
```

### 2. getOperationContext - 获取操作上下文

**签名**：`OperationContext getOperationContext()`

**说明**：获取操作的上下文信息，包含当前用户、组织、操作标识等运行时环境数据。

**业务场景**：保存操作时，根据当前登录用户所属组织来设置不同的审批策略。

```typescript
import { AbstractOperationServicePlugIn } from "@cosmic/bos-core/kd/bos/entity/plugin";

class OrgStrategyPlugin extends AbstractOperationServicePlugIn {

  beforeExecuteOperationTransaction(e: BeforeOperationArgs): void {
    super.beforeExecuteOperationTransaction(e);

    // 获取操作上下文
    let context = this.getOperationContext();

    // 获取当前用户ID
    let currentUserId = context.getUserId();
    // 获取当前操作的组织ID
    let orgId = context.getOrgId();
    // 获取操作标识
    let operationKey = context.getOperationKey();

    console.log("当前用户：" + currentUserId
      + "，组织：" + orgId
      + "，操作：" + operationKey);

    let validExtDatas = e.getValidExtDataEntities();
    for (let i = 0; i < validExtDatas.size(); i++) {
      let extData = validExtDatas.get(i);
      let dataEntity = extData.getDataEntity();

      // 根据组织设置不同的审批策略
      if (orgId == 100001) {
        dataEntity.set("kdec_approvestrategy", "fast_approve");
      } else {
        dataEntity.set("kdec_approvestrategy", "standard_approve");
      }
    }
  }
}

let plugin = new OrgStrategyPlugin();
export { plugin };
```

### 3. setOperationContext - 设置操作上下文

**签名**：`void setOperationContext(OperationContext operationContext)`

**说明**：由框架在操作初始化时调用，设置操作上下文。一般不需要在插件中主动调用，但在特殊场景（如手动构造操作调用链）中可使用。

**业务场景**：在自定义调度任务中手动执行操作时，需要设置操作上下文来模拟用户操作环境。

```typescript
import { AbstractOperationServicePlugIn } from "@cosmic/bos-core/kd/bos/entity/plugin";
import { RequestContext } from "@cosmic/bos-core/kd/bos/context";

class ContextAwarePlugin extends AbstractOperationServicePlugIn {

  // 说明：setOperationContext 通常由框架自动调用，
  // 在插件中一般通过 getOperationContext 获取已设置的上下文
  beginOperationTransaction(e: BeginOperationTransactionArgs): void {
    super.beginOperationTransaction(e);

    // 实际开发中，读取上下文获取用户信息
    let context = this.getOperationContext();
    let userId = context.getUserId();

    let dataEntities = e.getDataEntities();
    for (let dataEntity of dataEntities) {
      // 记录操作人信息
      dataEntity.set("kdec_operator", userId);
      dataEntity.set("kdec_operatetime", new Date());
    }
  }
}

let plugin = new ContextAwarePlugin();
export { plugin };
```

### 4. getDataEntities - 获取数据实体数组

**签名**：`DynamicObject[] getDataEntities()`

**说明**：获取当前操作涉及的所有数据实体数组，是操作插件中访问单据数据的核心方法。

**业务场景**：提交操作时，遍历所有待提交的采购订单，汇总计算总金额并记录日志。

```typescript
import { AbstractOperationServicePlugIn } from "@cosmic/bos-core/kd/bos/entity/plugin";
import { DynamicObject, DynamicObjectCollection } from "@cosmic/bos-core/kd/bos/dataentity/entity";

class SubmitAmountSummaryPlugin extends AbstractOperationServicePlugIn {

  onPreparePropertys(e: $.kd.bos.entity.plugin.PreparePropertysEventArgs): void {
    super.onPreparePropertys(e);
    e.getFieldKeys().add("billno");
    e.getFieldKeys().add("billentry.amount");
  }

  afterExecuteOperationTransaction(e: AfterOperationArgs): void {
    super.afterExecuteOperationTransaction(e);

    // 获取所有参与操作的数据实体
    let dataEntities = this.getDataEntities();

    let totalAmount = BigDecimal.ZERO;
    for (let dataEntity of dataEntities) {
      let billno = dataEntity.getString("billno");
      let entries = dataEntity.getDynamicObjectCollection("billentry") as DynamicObjectCollection;

      let billAmount = BigDecimal.ZERO;
      for (let i = 0; i < entries.size(); i++) {
        let entry = entries.get(i) as DynamicObject;
        let amount = entry.get("amount") as BigDecimal;
        if (amount != null) {
          billAmount = billAmount.add(amount);
        }
      }
      totalAmount = totalAmount.add(billAmount);
      console.log("单据 " + billno + " 金额合计：" + billAmount.toString());
    }

    console.log("本次提交总金额：" + totalAmount.toString());
  }
}

let plugin = new SubmitAmountSummaryPlugin();
export { plugin };
```

### 5. setDataEntities - 设置数据实体数组

**签名**：`void setDataEntities(DynamicObject[] dataEntities)`

**说明**：由框架在操作前调用以注入数据实体。一般不在插件中主动调用，但在需要替换或过滤操作数据时可使用。

**业务场景**：批量审核时，过滤掉金额为零的无效单据，仅对有效单据执行审核。

```typescript
import { AbstractOperationServicePlugIn } from "@cosmic/bos-core/kd/bos/entity/plugin";
import { DynamicObject, DynamicObjectCollection } from "@cosmic/bos-core/kd/bos/dataentity/entity";

class FilterZeroAmountPlugin extends AbstractOperationServicePlugIn {

  onPreparePropertys(e: $.kd.bos.entity.plugin.PreparePropertysEventArgs): void {
    super.onPreparePropertys(e);
    e.getFieldKeys().add("billno");
    e.getFieldKeys().add("kdec_totalamount");
  }

  beforeExecuteOperationTransaction(e: BeforeOperationArgs): void {
    super.beforeExecuteOperationTransaction(e);

    // 过滤金额为零的单据
    let validExtDatas = e.getValidExtDataEntities();
    let filteredList = new ArrayList();

    for (let i = 0; i < validExtDatas.size(); i++) {
      let extData = validExtDatas.get(i);
      let dataEntity = extData.getDataEntity();
      let totalAmount = dataEntity.get("kdec_totalamount") as BigDecimal;

      if (totalAmount != null && totalAmount.compareTo(BigDecimal.ZERO) > 0) {
        filteredList.add(extData);
      } else {
        console.log("跳过金额为零的单据：" + dataEntity.getString("billno"));
      }
    }

    // 用过滤后的数据替换原有数据
    e.getValidExtDataEntities().clear();
    e.getValidExtDataEntities().addAll(filteredList);
  }
}

let plugin = new FilterZeroAmountPlugin();
export { plugin };
```

### 6. setOperationResult - 设置操作结果

**签名**：`void setOperationResult(OperationResult operationResult)`

**说明**：由框架调用初始化操作结果对象。在插件中一般通过 `getOperationResult()` 获取结果并追加错误信息，而非直接替换整个结果对象。

**业务场景**：在操作执行后，向操作结果中追加自定义成功提示信息。

```typescript
import { AbstractOperationServicePlugIn } from "@cosmic/bos-core/kd/bos/entity/plugin";

class CustomResultPlugin extends AbstractOperationServicePlugIn {

  onPreparePropertys(e: $.kd.bos.entity.plugin.PreparePropertysEventArgs): void {
    super.onPreparePropertys(e);
    e.getFieldKeys().add("billno");
  }

  afterExecuteOperationTransaction(e: AfterOperationArgs): void {
    super.afterExecuteOperationTransaction(e);

    // 获取操作结果并追加自定义消息
    let result = this.getOperationResult();

    if (result.isSuccess()) {
      let dataEntities = e.getDataEntities();
      let billNos = new StringBuilder();
      for (let dataEntity of dataEntities) {
        billNos.append(dataEntity.getString("billno")).append(", ");
      }
      let message = "以下单据已成功处理：" + billNos.toString();
      result.setMessage(message);
    }
  }
}

let plugin = new CustomResultPlugin();
export { plugin };
```

### 7. setContext - 设置上下文

**签名**：`default void setContext(MainEntityType billEntityType, Map<String, Object> operateMeta, ...)`

**说明**：由框架调用，传入单据实体类型、操作元数据等上下文信息。一般不需要在插件中主动调用。

**业务场景**：通过操作上下文获取当前实体类型标识，实现一个插件同时服务多个单据实体的通用逻辑。

```typescript
import { AbstractOperationServicePlugIn } from "@cosmic/bos-core/kd/bos/entity/plugin";
import { InitOperationArgs } from "@cosmic/bos-core/kd/bos/entity/plugin/args";
import { QueryServiceHelper } from "@cosmic/bos-core/kd/bos/servicehelper";
import { QFilter, QCP } from "@cosmic/bos-core/kd/bos/orm/query";

class MultiEntityPlugin extends AbstractOperationServicePlugIn {

  onPreparePropertys(e: $.kd.bos.entity.plugin.PreparePropertysEventArgs): void {
    super.onPreparePropertys(e);
    e.getFieldKeys().add("billno");
    e.getFieldKeys().add("kdec_totalamount");
  }

  beforeExecuteOperationTransaction(e: BeforeOperationArgs): void {
    super.beforeExecuteOperationTransaction(e);

    // 通过操作上下文获取当前实体标识
    let context = this.getOperationContext();
    let entityNumber = context.getEntityNumber();

    // 根据不同实体执行不同校验逻辑
    let validExtDatas = e.getValidExtDataEntities();
    for (let i = 0; i < validExtDatas.size(); i++) {
      let extData = validExtDatas.get(i);
      let dataEntity = extData.getDataEntity();

      if (entityNumber == "pm_purorderbill") {
        // 采购订单：校验金额不超过预算
        let amount = dataEntity.get("kdec_totalamount") as BigDecimal;
        let budget = new BigDecimal("1000000");
        if (amount != null && amount.compareTo(budget) > 0) {
          e.setCancelMessage("采购订单金额超过预算上限100万");
          e.setCancel(true);
          return;
        }
      } else if (entityNumber == "sm_salorderbill") {
        // 销售订单：校验客户信用额度
        console.log("执行销售订单信用额度校验");
      }
    }
  }
}

let plugin = new MultiEntityPlugin();
export { plugin };
```

### 8. getOption - 获取操作选项

**签名**：`OperateOption getOption()`

**说明**：获取操作选项对象，可用于在事件之间传递自定义变量（通过 `setVariableValue`/`getVariableValue`），也可以读取调用方传入的自定义参数。

**业务场景**：提交操作时在 `beginOperationTransaction` 中生成关联单据，将关联单据ID通过 `Option` 传递给 `rollbackOperation`，以便失败时回滚。

```typescript
import { AbstractOperationServicePlugIn } from "@cosmic/bos-core/kd/bos/entity/plugin";
import { BusinessDataServiceHelper } from "@cosmic/bos-core/kd/bos/servicehelper";
import { SaveServiceHelper } from "@cosmic/bos-core/kd/bos/servicehelper/operation";
import { DynamicObject } from "@cosmic/bos-core/kd/bos/dataentity/entity";
import { OperateOption } from "@cosmic/bos-core/kd/bos/dataentity";
import { RequestContext } from "@cosmic/bos-core/kd/bos/context";

class OptionPassingPlugin extends AbstractOperationServicePlugIn {

  onPreparePropertys(e: $.kd.bos.entity.plugin.PreparePropertysEventArgs): void {
    super.onPreparePropertys(e);
    e.getFieldKeys().add("billno");
    e.getFieldKeys().add("kdec_supplier");
  }

  beginOperationTransaction(e: BeginOperationTransactionArgs): void {
    super.beginOperationTransaction(e);

    let dataEntities = e.getDataEntities();
    let createdIds = new StringBuilder();

    for (let dataEntity of dataEntities) {
      // 生成关联的收货通知单
      let noticeObj = BusinessDataServiceHelper.newDynamicObject("kdec_receivenotice") as DynamicObject;
      noticeObj.set("billstatus", "A");
      noticeObj.set("kdec_sourcebillno", dataEntity.getString("billno"));
      noticeObj.set("creator", RequestContext.get().getCurrUserId());

      let result = SaveServiceHelper.saveOperate(
        "kdec_receivenotice", [noticeObj], OperateOption.create()
      );

      if (result.isSuccess()) {
        let pkIds = result.getSuccessPkIds();
        pkIds.forEach(function(pkId: any) {
          createdIds.append(pkId).append(",");
        });
      }
    }

    // 通过 Option 传递已创建的关联单据ID，供 rollbackOperation 使用
    this.getOption().setVariableValue("created_notice_ids", createdIds.toString());
  }

  rollbackOperation(e: RollbackOperationArgs): void {
    super.rollbackOperation(e);

    // 从 Option 获取需要回滚的关联单据ID
    let createdIdsStr = this.getOption().getVariableValue("created_notice_ids") as string;
    if (createdIdsStr != null && createdIdsStr.length > 0) {
      console.log("事务回滚，需要清理已创建的收货通知单：" + createdIdsStr);
      // 执行关联单据的删除补偿
    }
  }
}

let plugin = new OptionPassingPlugin();
export { plugin };
```

### 9. setOperateProgress - 设置操作进度

**签名**：`default void setOperateProgress(OperateProgress progress)`

**说明**：设置操作执行进度对象，用于批量操作时向前端报告处理进度。

**业务场景**：批量审核大量采购订单时，设置进度条让用户了解审核进展。

```typescript
import { AbstractOperationServicePlugIn } from "@cosmic/bos-core/kd/bos/entity/plugin";

class BatchAuditProgressPlugin extends AbstractOperationServicePlugIn {

  onPreparePropertys(e: $.kd.bos.entity.plugin.PreparePropertysEventArgs): void {
    super.onPreparePropertys(e);
    e.getFieldKeys().add("billno");
    e.getFieldKeys().add("billentry.amount");
  }

  beginOperationTransaction(e: BeginOperationTransactionArgs): void {
    super.beginOperationTransaction(e);

    let dataEntities = e.getDataEntities();
    let total = dataEntities.length;

    // 获取进度对象并更新进度
    let progress = this.getOperateProgress();
    if (progress != null) {
      progress.setTotalCount(total);
    }

    for (let i = 0; i < total; i++) {
      let dataEntity = dataEntities[i];
      let billno = dataEntity.getString("billno");

      // 执行业务逻辑：计算单据金额
      console.log("正在处理第 " + (i + 1) + "/" + total + " 条：" + billno);

      // 更新进度
      if (progress != null) {
        progress.setCurrentCount(i + 1);
        progress.setMessage("正在审核：" + billno);
      }
    }
  }
}

let plugin = new BatchAuditProgressPlugin();
export { plugin };
```

### 11. isSupportSplitPageMode - 是否支持分页模式

**签名**：`default boolean isSupportSplitPageMode()`

**说明**：返回插件是否支持分页执行模式。当批量操作数据量很大时，系统会自动分页处理数据，避免一次性加载过多数据导致内存溢出。默认返回 `true`。

**业务场景**：批量删除基础资料时，由于需要全量数据进行引用检查，不能分页，需要关闭分页模式。

```typescript
import { AbstractOperationServicePlugIn } from "@cosmic/bos-core/kd/bos/entity/plugin";
import { QueryServiceHelper } from "@cosmic/bos-core/kd/bos/servicehelper";
import { QFilter, QCP } from "@cosmic/bos-core/kd/bos/orm/query";

class DeleteRefCheckPlugin extends AbstractOperationServicePlugIn {

  // 关闭分页模式，确保一次性获取所有待删除数据进行引用检查
  isSupportSplitPageMode(): boolean {
    return false;
  }

  onPreparePropertys(e: $.kd.bos.entity.plugin.PreparePropertysEventArgs): void {
    super.onPreparePropertys(e);
    e.getFieldKeys().add("number");
    e.getFieldKeys().add("name");
  }

  beforeExecuteOperationTransaction(e: BeforeOperationArgs): void {
    super.beforeExecuteOperationTransaction(e);

    // 获取所有待删除的基础资料编码
    let validExtDatas = e.getValidExtDataEntities();
    let numbers = new ArrayList();

    for (let i = 0; i < validExtDatas.size(); i++) {
      let extData = validExtDatas.get(i);
      let dataEntity = extData.getDataEntity();
      numbers.add(dataEntity.getString("number"));
    }

    // 一次性检查这些基础资料是否被单据引用
    let qFilter = new QFilter("kdec_material.number", QCP.in, numbers);
    let exists = QueryServiceHelper.exists("pm_purorderbill", [qFilter]);

    if (exists) {
      e.setCancelMessage("所选基础资料已被采购订单引用，不能删除");
      e.setCancel(true);
    }
  }
}

let plugin = new DeleteRefCheckPlugin();
export { plugin };
```

### 12. initialize - 初始化操作环境

**签名**：`default void initialize(InitializeOperationArgs e)`

**触发时机**：正式操作前准备环境，在所有事件之前最先执行。

**说明**：用于在操作执行前做一些环境准备工作，如根据操作类型决定是否继续执行。

**业务场景**：采购订单操作前，检查系统参数配置，决定是否允许执行当前操作。

```typescript
import { AbstractOperationServicePlugIn } from "@cosmic/bos-core/kd/bos/entity/plugin";
import { QueryServiceHelper } from "@cosmic/bos-core/kd/bos/servicehelper";
import { QFilter, QCP } from "@cosmic/bos-core/kd/bos/orm/query";

class InitCheckPlugin extends AbstractOperationServicePlugIn {

  initialize(e: InitOperationArgs): void {
    super.initialize(e);

    // InitOperationArgs 当前声明层没有暴露 getOperationKey，读取操作编码要走操作上下文
    let operationKey = this.getOperationContext().getOperationKey();
    console.log("当前操作：" + operationKey);

    // 查询系统参数，检查当前操作是否在维护期间被禁用
    let paramFilter = [new QFilter("number", QCP.equals, "SYSTEM_MAINTENANCE")];
    let sysParam = QueryServiceHelper.queryOne(
```

### 15. onAddValidators - 添加自定义校验器

**签名**：`default void onAddValidators(AddValidatorsEventArgs e)`

**触发时机**：系统预置校验器加载完毕，执行校验前。

**说明**：添加自定义校验器对单据数据进行校验，校验失败的单据将被剔除，不进入后续事务处理。校验器继承 `AbstractValidator`，支持批量校验，一条失败不影响其他数据。

**业务场景**：采购申请单提交时，校验每个分录行的数量必须大于零，且物料不能重复。

```typescript
import { AbstractOperationServicePlugIn, AddValidatorsEventArgs } from "@cosmic/bos-core/kd/bos/entity/plugin";
import { AbstractValidator, ErrorLevel, ValidationErrorInfo } from "@cosmic/bos-core/kd/bos/entity/validate";
import { RowDataModel } from "@cosmic/bos-core/kd/bos/entity/formula";
import { DynamicObjectCollection } from "@cosmic/bos-core/kd/bos/dataentity/metadata/dynamicobject";
import { LocaleString } from "@cosmic/bos-core/kd/bos/dataentity/entity";

class ReqSubmitValidatePlugin extends AbstractOperationServicePlugIn {

  onPreparePropertys(e: $.kd.bos.entity.plugin.PreparePropertysEventArgs): void {
    super.onPreparePropertys(e);
    e.getFieldKeys().add("billno");
    e.getFieldKeys().add("kdec_reqentry.kdec_material");
    e.getFieldKeys().add("kdec_reqentry.kdec_qty");
  }

  onAddValidators(e: AddValidatorsEventArgs): void {
    super.onAddValidators(e);

    // 添加自定义校验器
    let ValidatorClass = this.getValidatorClass();
    e.addValidator(new ValidatorClass());
  }

  getValidatorClass() {
    class ReqEntryValidator extends AbstractValidator {
      validate(): void {
        let extDataEntities = this.getDataEntities();

        for (let rowExtData of extDataEntities) {
          let dataEntity = rowExtData.getDataEntity();
          let entries = dataEntity.get("kdec_reqentry") as DynamicObjectCollection;

          // 校验1：分录不能为空
          if (entries == null || entries.size() == 0) {
            let msg = new LocaleString(rowExtData.getBillNo() + " 采购分录不能为空");
            let info = new ValidationErrorInfo(
              "", rowExtData.getBillPkId(),
              rowExtData.getDataEntityIndex(), rowExtData.getRowIndex(),
              "REQ_ENTRY_EMPTY",
              this.getValidateContext().getOperateName(),
              msg.toString(), ErrorLevel.Error
            );
            this.getValidateResult().addErrorInfo(info);
            continue;
          }

          // 校验2：每行数量必须大于零
          let materialSet = new HashSet();
          for (let i = 0; i < entries.size(); i++) {
            let entry = entries.get(i);
            let qty = entry.get("kdec_qty") as BigDecimal;

            if (qty == null || qty.compareTo(BigDecimal.ZERO) <= 0) {
              let msg = new LocaleString(
                rowExtData.getBillNo() + " 第" + (i + 1) + "行数量必须大于零"
              );
              let info = new ValidationErrorInfo(
                "", rowExtData.getBillPkId(),
                rowExtData.getDataEntityIndex(), i,
                "REQ_QTY_INVALID",
                this.getValidateContext().getOperateName(),
                msg.toString(), ErrorLevel.Error
              );
              this.getValidateResult().addErrorInfo(info);
            }

            // 校验3：物料不能重复
            let materialObj = entry.getDynamicObject("kdec_material");
            if (materialObj != null) {
              let materialId = materialObj.getLong("id");
              if (materialSet.contains(materialId)) {
                let msg = new LocaleString(
                  rowExtData.getBillNo() + " 第" + (i + 1) + "行物料重复"
                );
                let info = new ValidationErrorInfo(
                  "", rowExtData.getBillPkId(),
                  rowExtData.getDataEntityIndex(), i,
                  "REQ_MATERIAL_DUP",
                  this.getValidateContext().getOperateName(),
                  msg.toString(), ErrorLevel.Error
                );
                this.getValidateResult().addErrorInfo(info);
              }
              materialSet.add(materialId);
            }
          }
        }
      }
    }
    return ReqEntryValidator;
  }
}

let plugin = new ReqSubmitValidatePlugin();
export { plugin };
```

### 16. beforeExecuteOperationTransaction - 事务前处理

### 17. beginOperationTransaction - 事务内执行

**签名**：`default void beginOperationTransaction(BeginOperationTransactionArgs e)`

**触发时机**：数据库事务已开启，数据提交入库前。

**说明**：在事务范围内处理关联数据同步。不支持跨库写操作；批量处理数据时一条失败则全部失败（事务回滚）。如有向第三方系统同步数据的需求，需在 `rollbackOperation` 中做补偿。

**业务场景**：采购申请单提交时，在同一事务内自动生成对应的采购订单。

```typescript
import { AbstractOperationServicePlugIn } from "@cosmic/bos-core/kd/bos/entity/plugin";
import { BusinessDataServiceHelper } from "@cosmic/bos-core/kd/bos/servicehelper";
import { SaveServiceHelper } from "@cosmic/bos-core/kd/bos/servicehelper/operation";
import { DynamicObject, DynamicObjectCollection } from "@cosmic/bos-core/kd/bos/dataentity/entity";
import { OperateOption } from "@cosmic/bos-core/kd/bos/dataentity";
import { OperationResult, OperateErrorInfo } from "@cosmic/bos-core/kd/bos/entity/operate/result";
import { KDException, ErrorCode } from "@cosmic/bos-core/kd/bos/exception";
import { RequestContext } from "@cosmic/bos-core/kd/bos/context";
import { QFilter } from "@cosmic/bos-core/kd/bos/orm/query";

class ReqSubmitCreateOrderPlugin extends AbstractOperationServicePlugIn {

  onPreparePropertys(e: $.kd.bos.entity.plugin.PreparePropertysEventArgs): void {
    super.onPreparePropertys(e);
    e.getFieldKeys().add("billno");
    e.getFieldKeys().add("kdec_supplier");
    e.getFieldKeys().add("kdec_department");
    e.getFieldKeys().add("kdec_reqentry.kdec_material");
    e.getFieldKeys().add("kdec_reqentry.kdec_qty");
    e.getFieldKeys().add("kdec_reqentry.kdec_price");
    e.getFieldKeys().add("kdec_orderno");
  }

  beginOperationTransaction(e: BeginOperationTransactionArgs): void {
    super.beginOperationTransaction(e);

    // 获取人民币币种
    let currencyObj = BusinessDataServiceHelper.loadSingle(
      "bd_currency", [new QFilter("number", "=", "CNY")]
    );

    let orderDojs: DynamicObject[] = [];
    let dataEntities = e.getDataEntities();

    for (let dataEntity of dataEntities) {
      // 创建采购订单
      let orderDoj = BusinessDataServiceHelper.newDynamicObject("pm_purorderbill") as DynamicObject;
      orderDoj.set("billstatus", "A");
      orderDoj.set("creator", RequestContext.get().getCurrUserId());
      orderDoj.set("kdec_supplier", dataEntity.get("kdec_supplier"));
      orderDoj.set("kdec_department", dataEntity.get("kdec_department"));
      orderDoj.set("kdec_currency", currencyObj);
      orderDoj.set("kdec_sourcebillno", dataEntity.getString("billno"));

      // 复制分录数据
      let reqEntries = dataEntity.getDynamicObjectCollection("kdec_reqentry") as DynamicObjectCollection;
      let orderEntries = orderDoj.getDynamicObjectCollection("billentry") as DynamicObjectCollection;

      for (let i = 0; i < reqEntries.size(); i++) {
        let reqEntry = reqEntries.get(i) as DynamicObject;
        let orderEntry = orderEntries.addNew() as DynamicObject;
        orderEntry.set("seq", i + 1);
        orderEntry.set("kdec_material", reqEntry.get("kdec_material"));
        orderEntry.set("kdec_qty", reqEntry.get("kdec_qty"));
        orderEntry.set("kdec_price", reqEntry.get("kdec_price"));

        let qty = reqEntry.get("kdec_qty") as BigDecimal;
        let price = reqEntry.get("kdec_price") as BigDecimal;
        if (qty != null && price != null) {
          orderEntry.set("kdec_amount", qty.multiply(price).setScale(2, BigDecimal.ROUND_HALF_UP));
        }
      }

      orderDojs.push(orderDoj);
    }

    // 保存采购订单
    let result = SaveServiceHelper.saveOperate(
      "pm_purorderbill", orderDojs, OperateOption.create()
    );

    if (result.isSuccess()) {
      // 将生成的订单编号回写到申请单
      let successPkIds = result.getSuccessPkIds();
      let orderIds = new StringBuilder();
      successPkIds.forEach(function(pkId: any) {
        orderIds.append(pkId).append(",");
      });
      this.getOption().setVariableValue("created_order_ids", orderIds.toString());

      for (let idx = 0; idx < dataEntities.length; idx++) {
        dataEntities[idx].set("kdec_orderno", orderDojs[idx].getString("billno"));
      }
    } else {
      let errorInfos = result.getAllErrorOrValidateInfo();
      let errMsg = "生成采购订单失败";
      if (errorInfos.size() > 0) {
        let errorInfo = errorInfos.get(0) as OperateErrorInfo;
        errMsg = errMsg + "：" + errorInfo.getMessage();
      }
      let errorCode = new ErrorCode("kdec_req.create_order_fail", errMsg);
      throw new KDException(errorCode, []);
    }
  }
}
```

### 19. rollbackOperation - 事务回滚补偿

**签名**：`default void rollbackOperation(RollbackOperationArgs e)`

**触发时机**：事务提交失败并回滚后。

**说明**：处理非事务性操作的补偿逻辑。如果在 `beginOperationTransaction` 或 `endOperationTransaction` 中有同步数据到第三方系统等非事务操作，需要在此方法中做回滚补偿。

**业务场景**：采购订单审核时同步数据到外部 ERP 系统，若苍穹事务回滚，则调用外部系统接口删除已同步的数据。

```typescript
import { AbstractOperationServicePlugIn } from "@cosmic/bos-core/kd/bos/entity/plugin";
import { DynamicObject } from "@cosmic/bos-core/kd/bos/dataentity/entity";

class ThirdPartyRollbackPlugin extends AbstractOperationServicePlugIn {

  onPreparePropertys(e: $.kd.bos.entity.plugin.PreparePropertysEventArgs): void {
    super.onPreparePropertys(e);
    e.getFieldKeys().add("billno");
    e.getFieldKeys().add("kdec_supplier.name");
    e.getFieldKeys().add("kdec_totalamount");
  }

  beginOperationTransaction(e: BeginOperationTransactionArgs): void {
    super.beginOperationTransaction(e);

    let dataEntities = e.getDataEntities();
    let syncedBillNos = new StringBuilder();

    for (let dataEntity of dataEntities) {
      let billno = dataEntity.getString("billno");

      // 模拟同步数据到第三方 ERP 系统（HTTP 调用等）
      let syncSuccess = this.syncToExternalSystem(dataEntity);

      if (syncSuccess) {
        syncedBillNos.append(billno).append(",");
      }
    }

    // 记录已同步的单据编号，供回滚时补偿使用
    this.getOption().setVariableValue("synced_billnos", syncedBillNos.toString());
  }

  rollbackOperation(e: RollbackOperationArgs): void {
    super.rollbackOperation(e);

    // 获取已同步到第三方系统的单据编号
    let syncedBillNos = this.getOption().getVariableValue("synced_billnos") as string;

    if (syncedBillNos != null && syncedBillNos.length > 0) {
      let billNos = syncedBillNos.split(",");
      for (let billno of billNos) {
        if (billno.length > 0) {
          // 调用第三方系统接口删除已同步的数据
          console.log("回滚补偿：删除第三方系统中已同步的单据 " + billno);
          this.deleteFromExternalSystem(billno);
        }
      }
    }

    // 记录回滚日志
    let rollbackEntities = e.getDataEntitys();
    for (let dataEntity of rollbackEntities) {
      console.log("事务回滚，单据：" + dataEntity.getString("billno"));
    }
  }

  // 模拟同步到第三方系统
  syncToExternalSystem(dataEntity: DynamicObject): boolean {
    console.log("同步单据到第三方 ERP：" + dataEntity.getString("billno"));
    return true;
  }

  // 模拟从第三方系统删除
  deleteFromExternalSystem(billno: string): void {
    console.log("从第三方 ERP 删除单据：" + billno);
  }
}

let plugin = new ThirdPartyRollbackPlugin();
```

### 20. afterExecuteOperationTransaction - 事务后处理

**签名**：`default void afterExecuteOperationTransaction(AfterOperationArgs e)`

**触发时机**：操作执行完毕，事务已提交。

```typescript
      messageInfo.setContentUrl(
        clientUrl + "?formId=kdec_reqbill&pkid=" + dataEntity.getPkValue()
      );

      messageInfos.add(messageInfo);
    }

    if (messageInfos.size() > 0) {
      let result = MessageCenterServiceHelper.batchSendMessages(messageInfos);
      console.log("审批通知发送完成，数量：" + messageInfos.size());
    }
  }
```

### 21. onReturnOperation - 返回操作结果

**说明**：在操作结果返回给调用方之前的最后一次处理机会，可以修改返回消息、追加返回数据。适合在此做操作完成的收尾工作和资源释放。

**业务场景**：批量操作完成后，汇总成功与失败的统计信息，返回给前端展示。

```typescript
import { AbstractOperationServicePlugIn } from "@cosmic/bos-core/kd/bos/entity/plugin";

class OperationSummaryPlugin extends AbstractOperationServicePlugIn {

  onPreparePropertys(e: $.kd.bos.entity.plugin.PreparePropertysEventArgs): void {
    super.onPreparePropertys(e);
    e.getFieldKeys().add("billno");

    let result = this.getOperationResult();
    let successCount = result.getSuccessPkIds().size();

    // 获取错误信息
    let errorInfos = result.getAllErrorOrValidateInfo();
    let failCount = errorInfos.size();

    // 构造汇总消息
    let summary = new StringBuilder();
    summary.append("操作完成：");
    summary.append("成功 " + successCount + " 条");

    if (failCount > 0) {
      summary.append("，失败 " + failCount + " 条");
      summary.append("\n失败明细：");
      for (let i = 0; i < errorInfos.size(); i++) {
        let errorInfo = errorInfos.get(i);
        summary.append("\n  - " + errorInfo.getMessage());
      }
    }

    // 设置返回消息
    result.setMessage(summary.toString());
    console.log(summary.toString());
  }
}
```

### 23. validatePrefix - 验证编号前缀

```typescript
import { AbstractOperationServicePlugIn } from "@cosmic/bos-core/kd/bos/entity/plugin";
```

## 完整生命周期综合案例

以下是一个完整的操作插件，展示所有生命周期方法在同一个业务流程中的协作：

**场景**：采购申请单自定义提交操作，包括字段预加载、数据校验、金额计算、生成采购订单、库存预留、消息通知、异常回滚等完整流程。

```typescript
import { AbstractOperationServicePlugIn, AddValidatorsEventArgs } from "@cosmic/bos-core/kd/bos/entity/plugin";
import { AbstractValidator, ErrorLevel, ValidationErrorInfo } from "@cosmic/bos-core/kd/bos/entity/validate";
import { BusinessDataServiceHelper, QueryServiceHelper } from "@cosmic/bos-core/kd/bos/servicehelper";
import { SaveServiceHelper } from "@cosmic/bos-core/kd/bos/servicehelper/operation";
import { DynamicObject, DynamicObjectCollection, LocaleString } from "@cosmic/bos-core/kd/bos/dataentity/entity";
import { OperateOption } from "@cosmic/bos-core/kd/bos/dataentity";
import { OperationResult } from "@cosmic/bos-core/kd/bos/entity/operate/result";
import { KDException, ErrorCode } from "@cosmic/bos-core/kd/bos/exception";
import { MessageCenterServiceHelper } from "@cosmic/bos-core/kd/bos/servicehelper/workflow";
import { MessageInfo } from "@cosmic/bos-core/kd/bos/workflow/engine/msg/info";
import { RequestContext } from "@cosmic/bos-core/kd/bos/context";
import { QFilter, QCP } from "@cosmic/bos-core/kd/bos/orm/query";

class ReqBillFullLifecyclePlugin extends AbstractOperationServicePlugIn {

  // ========== 1. 字段预加载 ==========
  onPreparePropertys(e: $.kd.bos.entity.plugin.PreparePropertysEventArgs): void {
    super.onPreparePropertys(e);
    let fields = e.getFieldKeys();
    fields.add("billno");
    fields.add("kdec_supplier");
    fields.add("kdec_supplier.name");
    fields.add("kdec_department");
    fields.add("kdec_totalamount");
    fields.add("kdec_reqentry.kdec_material");
    fields.add("kdec_reqentry.kdec_qty");
    fields.add("kdec_reqentry.kdec_price");
    fields.add("kdec_reqentry.kdec_amount");
    fields.add("kdec_orderno");
  }

  // ========== 2. 自定义校验器 ==========
  onAddValidators(e: AddValidatorsEventArgs): void {
    super.onAddValidators(e);
    let ValidatorClass = this.getValidatorClass();
    e.addValidator(new ValidatorClass());
  }

  getValidatorClass() {
    class ReqValidator extends AbstractValidator {
      validate(): void {
        let extDataEntities = this.getDataEntities();
        for (let rowExtData of extDataEntities) {
          let dataEntity = rowExtData.getDataEntity();
          let entries = dataEntity.get("kdec_reqentry") as DynamicObjectCollection;
          if (entries == null || entries.size() == 0) {
            let msg = new LocaleString(rowExtData.getBillNo() + " 分录不能为空");
            let info = new ValidationErrorInfo(
              "", rowExtData.getBillPkId(),
              rowExtData.getDataEntityIndex(), rowExtData.getRowIndex(),
              "ENTRY_EMPTY", this.getValidateContext().getOperateName(),
              msg.toString(), ErrorLevel.Error
            );
            this.getValidateResult().addErrorInfo(info);
          }
        }
      }
    }
    return ReqValidator;
  }

  // ========== 3. 事务前：金额计算（耗时逻辑放事务外） ==========
  beforeExecuteOperationTransaction(e: BeforeOperationArgs): void {
    super.beforeExecuteOperationTransaction(e);

    let validExtDatas = e.getValidExtDataEntities();
    for (let i = 0; i < validExtDatas.size(); i++) {
      let extData = validExtDatas.get(i);
      let dataEntity = extData.getDataEntity();

      // 重新计算分录金额
      let entries = dataEntity.getDynamicObjectCollection("kdec_reqentry") as DynamicObjectCollection;
      let totalAmount = BigDecimal.ZERO;
      for (let j = 0; j < entries.size(); j++) {
        let entry = entries.get(j) as DynamicObject;
        let qty = entry.get("kdec_qty") as BigDecimal;
        let price = entry.get("kdec_price") as BigDecimal;
        if (qty != null && price != null) {
          let amount = qty.multiply(price).setScale(2, BigDecimal.ROUND_HALF_UP);
          entry.set("kdec_amount", amount);
          totalAmount = totalAmount.add(amount);
        }
      }
      dataEntity.set("kdec_totalamount", totalAmount);
    }
  }

  // ========== 4. 事务内：生成采购订单 ==========
  beginOperationTransaction(e: BeginOperationTransactionArgs): void {
    super.beginOperationTransaction(e);

    let dataEntities = e.getDataEntities();
    let orderDojs: DynamicObject[] = [];

    for (let dataEntity of dataEntities) {
      let orderDoj = BusinessDataServiceHelper.newDynamicObject("pm_purorderbill") as DynamicObject;
      orderDoj.set("billstatus", "A");
      orderDoj.set("creator", RequestContext.get().getCurrUserId());
      orderDoj.set("kdec_sourcebillno", dataEntity.getString("billno"));
      orderDojs.push(orderDoj);
    }

    let result = SaveServiceHelper.saveOperate("pm_purorderbill", orderDojs, OperateOption.create());

    if (result.isSuccess()) {
      let ids = new StringBuilder();
      result.getSuccessPkIds().forEach(function(pkId: any) {
        ids.append(pkId).append(",");
      });
      this.getOption().setVariableValue("order_ids", ids.toString());
    } else {
      let errorCode = new ErrorCode("req.submit.fail", "生成采购订单失败");
      throw new KDException(errorCode, []);
    }
  }

  // ========== 5. 数据入库后：关联数据更新 ==========
  endOperationTransaction(e: EndOperationTransactionArgs): void {
    super.endOperationTransaction(e);
    console.log("数据已入库，执行关联更新");
  }

  // ========== 6. 事务回滚：补偿处理 ==========
  rollbackOperation(e: RollbackOperationArgs): void {
    super.rollbackOperation(e);

    let orderIds = this.getOption().getVariableValue("order_ids") as string;
    if (orderIds != null && orderIds.length > 0) {
      console.log("事务回滚，需删除已生成的采购订单：" + orderIds);
    }
  }

  // ========== 7. 事务后：发送通知 ==========
  afterExecuteOperationTransaction(e: AfterOperationArgs): void {
    super.afterExecuteOperationTransaction(e);

    let userId = RequestContext.get().getCurrUserId();
    let dataEntities = e.getDataEntities();
    let messageInfos = new ArrayList();

    for (let dataEntity of dataEntities) {
      let messageInfo = new MessageInfo();
      messageInfo.setMessageTitle(new LocaleString("采购申请已提交：" + dataEntity.getString("billno")));
      messageInfo.setMessageContent(new LocaleString("请及时审批"));
      let receivers = new ArrayList();
      receivers.add(userId);
      messageInfo.setUserIds(receivers);
      messageInfo.setMessageType(MessageInfo.TYPE_MESSAGE);
      messageInfo.setSenderId(userId);
      messageInfo.setEntityNumber("kdec_reqbill");
      messageInfo.setBizDataId(dataEntity.getPkValue());
      messageInfos.add(messageInfo);
    }

    if (messageInfos.size() > 0) {
      MessageCenterServiceHelper.batchSendMessages(messageInfos);
    }
  }

  // ========== 8. 返回结果：汇总信息 ==========
  onReturnOperation(e: ReturnOperationArgs): void {
    super.onReturnOperation(e);

    let result = this.getOperationResult();
    if (result.isSuccess()) {
      result.setMessage("提交成功，已生成关联采购订单并发送审批通知");
    }
  }
}

let plugin = new ReqBillFullLifecyclePlugin();
export { plugin };
```
